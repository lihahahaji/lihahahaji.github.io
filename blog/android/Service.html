# 1. 简单服务

  1. 创建一个服务
package com.haji.test9;

import android.app.Notification;
import android.app.NotificationChannel;
import android.app.NotificationManager;
import android.app.Service;
import android.content.Context;
import android.content.Intent;
import android.os.IBinder;
import android.util.Log;
import android.view.View;


public class UpdateService extends Service {
    public UpdateService() {
//        Log.i("lihaji","UpdateService:OnCreate");
    }

    private Thread thread;

    @Override
    public IBinder onBind(Intent intent) {
        // TODO: Return the communication channel to the service.
        throw new UnsupportedOperationException("Not yet implemented");
    }

    @Override
    public void onCreate() {
        Log.i("lihaji","UpdateService:OnCreate");
        super.onCreate();
        thread=new Thread(MyThread);
        thread.start();
    }
    public void onDestroy()
    {
        super.onDestroy();
        thread.interrupt();
    }

    final Runnable MyThread = new Runnable()
    {
        public void run()
        {
            while(!Thread.interrupted())
            {
                try {
//                    Thread.sleep(10000);
//                    Log.i("THREAD","Thread is running !");
                    while(true)
                    {
                        Thread.sleep(1000);
                        Log.i("lihaji","有更新了！");
                        senNotification();
                    }
                } catch (InterruptedException e) {
                    Log.i("THREAD","Thread is interrupted !");
                    e.printStackTrace();
                    break;
                }

            }
        }
    };

    public void senNotification() {
        // 1. 获取系统通知管理器
        NotificationManager nm = (NotificationManager) getSystemService(Context.NOTIFICATION_SERVICE);
        Context context=getApplicationContext();

        // 2. 设置通知显示样式
        String channelId = "测试渠道";
        Notification notification = new Notification.Builder(context,channelId)
                .setContentTitle("有更新了！")
                .setContentText("有更新了！")
                .setWhen(System.currentTimeMillis())
                .setSmallIcon(R.drawable.ic_launcher_background)
//                .setLargeIcon(BitmapFactory.decodeResource(getResources(),R.drawable.welcome1))   //设置大图标
                .build();
        // 3. 建立通知渠道
        NotificationChannel channel = new NotificationChannel(channelId,"测试渠道名称", NotificationManager.IMPORTANCE_DEFAULT);
        nm.createNotificationChannel(channel);

        // 4. 发送通知
        nm.notify(1123, notification);
    }
}

# 2. 使用 bindService 来绑定服务

  1. 创建一个服务
package com.haji.test9;

import android.app.Service;
import android.content.Intent;
import android.media.MediaPlayer;
import android.os.Binder;
import android.os.IBinder;
import android.util.Log;

public class MusicService extends Service {

    public final IBinder m_mbinder = new MusicBinder();
    private MediaPlayer m_mp=null;

    public MusicService() {
    }

    @Override
    public IBinder onBind(Intent intent) {
        // TODO: Return the communication channel to the service.
//        throw new UnsupportedOperationException("Not yet implemented");
        Log.i("lihaji","MusicService:onBind");
        return m_mbinder;
    }

    public class MusicBinder extends Binder
    {
        // 用于得到 所属的主服务，便于调用主服务的各种函数
        MusicService getService()
        {
            return MusicService.this;
        }
    }



    public void onCreate()
    {
        super.onCreate();
        Log.i("lihaji","MusicService:OnCreate");
    }



    public void onDestroy()
    {
        super.onDestroy();
        m_mp.stop();
        Log.i("lihaji","MusicService:onDestroy");
    }

    public boolean onUnbind(Intent intent)
    {
        Log.i("lihaji","MusicService:onUnbind");
        return super.onUnbind(intent);

    }


    public void PlayMusic()
    {
        if(m_mp==null)
        {
            m_mp = MediaPlayer.create(this,R.raw.freedom);
            m_mp.setLooping(false);
        }
        if(!m_mp.isPlaying())
        {
            Log.i("lihaji","MUSICSTART");
            m_mp.start();
        }
    }

}
  1. 在 MainActivity 中绑定服务
// 1. 先建立一个 intent
Intent MusicServiceIntent=new Intent();
MusicServiceIntent.setClass(MainActivity.this,MusicService.class);
// 2. 创建一个 ServiceConnection
ServiceConnection sc = new ServiceConnection() {
    @Override
    public void onServiceConnected(ComponentName componentName, IBinder iBinder) {
        // 获取服务
        MusicService ms = ((MusicService.MusicBinder)iBinder).getService();
        if(ms != null)
        {
            ms.PlayMusic();
        }
    }

    @Override
    public void onServiceDisconnected(ComponentName componentName) {
        
    }
};

// 3. 绑定服务
bindService(MusicServiceIntent,sc,BIND_AUTO_CREATE);

// 4. 解除绑定
unbindService(sc);
