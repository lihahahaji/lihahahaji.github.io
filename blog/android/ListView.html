  1. 创建一个MyBaseAdapter继承BaseAdapter类，并且实现对ListView控件的数据适配
// 创建一个MyBaseAdapter继承BaseAdapter类，并且实现对ListView控件的数据适配
class MyBaseAdapter extends BaseAdapter
{
    String [] textLarge={"张三","李四","王五","林峰","吴凯","小猫"};
    String [] textSmall={"我是张三","我是李四","我是王五","我是笨比","我是啥子","我是小猫"};
    @Override
    public int getCount() { // 获取Items的总数
        return textLarge.length;
    }

    @Override
    public Object getItem(int position) { // 返回Item的数据对象
        return textLarge[position];
    }

    @Override
    public long getItemId(int position) {   // 返回Item的ID
        return position;
    }

    @Override
    public View getView(int position, View convertView, ViewGroup parent) {     // 返回Item的视图
        // 加载List_Item的布局文件
        View view =View.inflate(SimpleLV.this,R.layout.simple_li,null);

        TextView tvLarge=(TextView)view.findViewById(R.id.tv_LargeText);
        TextView tvSmall=(TextView)view.findViewById(R.id.tv_SmallText);
        ImageView imgV=(ImageView)view.findViewById(R.id.imgV_icon);
        
        imgV.setImageResource(R.mipmap.ic_launcher);// 设置头像的内容
        tvLarge.setText(textLarge[position]); // 设置大文本的内容
        tvSmall.setText(textSmall[position]); // 设置小文本的内容
        return view;
    }
}
  1. 创建 ListView 对象
ListView mListView=(ListView)findViewById(R.id.lv_ListViewTest);
MyBaseAdapter mAdapter=new MyBaseAdapter();
mListView.setAdapter(mAdapter);

mListView.setOnItemClickListener(new AdapterView.OnItemClickListener() {
    @Override
    public void onItemClick(AdapterView<?> parent, View view, int position, long id) {
        TextView tv=view.findViewById(R.id.txt_item_id);
        String text="你选中了"+tv.getText().toString();
        Toast.makeText(MyListView.this,text,Toast.LENGTH_SHORT).show();
    }
});
