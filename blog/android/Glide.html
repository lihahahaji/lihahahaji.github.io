# 使用 Glide 加载网络图片

# 1. 添加依赖

implementation 'com.github.bumptech.glide:glide:3.7.0'

# 2. 在 Manifest 中开启网络访问权限

<uses-permission android:name="android.permission.INTERNET"></uses-permission>

# 3. 调用 Glide

// 使用 Glide 获取网络图片
Glide.with(getContext())
        .load("https://www.bing.com/th?id=OHR.ElephantGiving_ZH-CN9743352473_1920x1080.jpg&rf=LaDigue_1920x1080.jpg&pid=hp")// 设置资源链接
        .error(R.drawable.rinbow) // 设置加载失败显示的图片
        .into(imageView); // 设置 加载的容器  一般为 imageView
