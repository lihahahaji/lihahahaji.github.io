# HelloCharts

引用 HelloCharts ： 在 Build.gradle(app) 中添加依赖即可

implementation 'com.github.lecho:hellocharts-library:1.5.8@aar'

# LineChartView

  1. 布局文件
<lecho.lib.hellocharts.view.LineChartView
    android:id="@+id/LCV_my_LineChartView"
    android:layout_width="match_parent"
    android:layout_height="288dp"
    app:layout_constraintStart_toStartOf="parent"
    app:layout_constraintTop_toTopOf="parent" />
  1. java 代码
//  初始化 LineChart 控件
public void initLineChart()
{

    // 添加坐标点的信息
    List<PointValue> values=new ArrayList<>();
    values.add(new PointValue(0, 2));
    values.add(new PointValue(1, 3));
    values.add(new PointValue(2, 3));
    values.add(new PointValue(3, 4));

    // 设置折线条样式
    Line line=new Line(values)
            .setColor(Color.BLUE) // 设置折线的颜色
            .setShape(ValueShape.CIRCLE) // 设置折线图上每个数据点的形状
            .setCubic(false) // 设置是否为曲线
            .setStrokeWidth(3) //设置线条的粗细，默认是3
            .setFilled(true) // 设置是否填充曲线覆盖的面积
            .setHasLabels(true) // 设置曲线的数据坐标是否加上备注
            .setFormatter(null) // 设置显示小数点
            .setHasLabelsOnlyForSelected(false) // 设置是否 点击数据坐标显示提示信息
            .setHasLines(true) // 设置是否有线条，如果设为 False 则只显示点
            .setHasPoints(true) // 设置是否有点，如果设置为 False 则只显示线没有点
            .setPointRadius(4); // 设置点的显示半径

    // 设置折线 列表 lines
    List<Line> lines=new ArrayList<>();

    // 添加到 lines
    lines.add(line);

    // 添加到 LineChartData 中
    LineChartData data=new LineChartData();
    data.setLines(lines);



    // 添加坐标轴

    // X轴
    Axis axisX = new Axis()
            .setName("X") // 设置轴的标注
            .setTextColor(Color.BLACK) // 设置标注文字的大小
            .setTextSize(7) // 设置标注文字的大小
            .setHasLines(false); // 设置X轴的分割线

    data.setAxisXBottom(axisX); //设置X轴在底部

    // Y轴
    Axis axisY = new Axis()
            .setName("Y") // 设置轴的标注
            .setTextColor(Color.BLACK) // 设置标注文字的大小
            .setTextSize(7) // 设置标注文字的大小
            .setHasLines(false); // 设置X轴的分割线
    data.setAxisYLeft(axisX); //设置X轴在左边

    

    // 部署到LineChartView 中
    LineChartView chart = findViewById(R.id.LCV_my_LineChartView);
    chart.setLineChartData(data);

}

# PieChartView

  1. 布局文件
<lecho.lib.hellocharts.view.PieChartView
    android:id="@+id/PCV_my_PieChartView"
    android:layout_width="match_parent"
    android:layout_height="300dp"
    android:layout_marginTop="308dp"
    app:layout_constraintTop_toTopOf="parent"
    tools:layout_editor_absoluteX="0dp" />
  1. java 代码
public void initPieChartView()
{

    // 获取PieChartView对象
    PieChartView chart = findViewById(R.id.PCV_my_PieChartView);
    chart.setViewportCalculationEnabled(true); // 设置饼图自动适应大小

    // 点击 Pie 时显示
    chart.setOnValueTouchListener(new PieChartOnValueSelectListener() {
        @Override
        public void onValueSelected(int arcIndex, SliceValue value) {
            Toast.makeText(MyHelloCharts.this,value.toString(),Toast.LENGTH_SHORT).show();
        }

        @Override
        public void onValueDeselected() {

        }
    });

    // 添加 数据
    List<SliceValue> sliceList =new ArrayList<SliceValue>();
    sliceList.add(new SliceValue(10,Color.RED).setLabel("Pie1"));
    sliceList.add(new SliceValue(20,Color.BLUE).setLabel("Pie2"));
    sliceList.add(new SliceValue(30,Color.YELLOW).setLabel("Pie3"));
    sliceList.add(new SliceValue(35,Color.GREEN).setLabel("Pie4"));
    sliceList.add(new SliceValue(5,Color.LTGRAY).setLabel("Pie5"));


    // 获取PieChartData对象
    PieChartData data=new PieChartData()
            .setValues(sliceList); // 设置数据

    // 为饼图设置数据
    chart.setPieChartData(data);

}
