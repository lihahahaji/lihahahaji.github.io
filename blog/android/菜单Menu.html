  2. 在 Activity 中重载 onCreateOptionsMenu 函数

public boolean onCreateOptionsMenu(Menu menu)
{
    // 方 法 getMenuInflater() 返回Activity的一个 MenuInflater ，使用这个对象，你可以调用 inflate(), 把菜单资源转换为菜单对象．
    getMenuInflater().inflate(R.menu.main,menu);
    return true;
}
  1. 定义 menu 文件中的内容 
<?xml version="1.0" encoding="utf-8"?>
<menu xmlns:android="http://schemas.android.com/apk/res/android">

    <item
        android:id="@+id/menu_m1"
        android:icon="@drawable/icon_6"
        android:title="item1"/>

</menu>
  1. 注意 android 4.0 以上默认选项菜单不显示 icon ，如何显示：
// 在使用选项菜单的 Activity 中添加如下代码
private void setIconEnable(Menu menu, boolean enable)
{
    try
    {
        Class<?> clazz = Class.forName("androidx.appcompat.view.menu.MenuBuilder");
        Method m = clazz.getDeclaredMethod("setOptionalIconsVisible", boolean.class);
        m.setAccessible(true);

        //MenuBuilder实现Menu接口，创建菜单时，传进来的menu其实就是MenuBuilder对象(java的多态特征)
        m.invoke(menu, enable);

    } catch (Exception e)
    {
        e.printStackTrace();
    }
}

// 并且在 onCreateOptionsMenu 中调用该方法
public boolean onCreateOptionsMenu(Menu menu)
{
    // 方 法 getMenuInflater() 返回Activity的一个 MenuInflater ，使用这个对象，你可以调用 inflate(), 把菜单资源转换为菜单对象．
    getMenuInflater().inflate(R.menu.main,menu);
    setIconEnable(menu,true);
    return true;
}
  1. 注册点击事件
// 重载函数 onOptionsItemSelected 来响应选项菜单的 item 点击事件
public boolean onOptionsItemSelected(MenuItem item)
{
    int id=item.getItemId();
    if(id==R.id.menu_m1) Toast.makeText(this,"item1",Toast.LENGTH_SHORT).show();
    else if(id==R.id.menu_m2) Toast.makeText(this,"item2",Toast.LENGTH_SHORT).show();
    else if(id==R.id.menu_m3) Toast.makeText(this,"item3",Toast.LENGTH_SHORT).show();
    else Toast.makeText(this,"items",Toast.LENGTH_SHORT).show();
    return super.onOptionsItemSelected(item);
}

java 代码1（以按下按钮弹出为例）

public void initButton1()
{

    // 1. 创建 PopupMenu 对象
    PopupMenu popupMenu =new PopupMenu(MainActivity.this,findViewById(R.id.btn_button));

    // 2. 调用 inflate 方法来将菜单资源填充到 PopupMenu 中
    popupMenu.inflate(R.menu.popupmenu_1);

    // 4. 注册 item 的点击事件
    popupMenu.setOnMenuItemClickListener(new PopupMenu.OnMenuItemClickListener() {
        @Override
        public boolean onMenuItemClick(MenuItem item) {
            int id=item.getItemId();
            if(id==R.id.subItem_Blue) ((TextView)findViewById(R.id.tv_Title)).setTextColor(Color.BLUE);
            else if(id==R.id.subItem_Green) ((TextView)findViewById(R.id.tv_Title)).setTextColor(Color.GREEN);
            else if(id==R.id.subItem_Red) ((TextView)findViewById(R.id.tv_Title)).setTextColor(Color.RED);
            return true;
        }
    });

    findViewById(R.id.btn_button).setOnClickListener(new View.OnClickListener() {
        @Override
        public void onClick(View v) {

            // 3. 调用 show 方法显示弹出菜单
            popupMenu.show();
        }
    });
}

# 上下文菜单

  1. 编写 XML 文件
<?xml version="1.0" encoding="utf-8"?>
<menu xmlns:android="http://schemas.android.com/apk/res/android">
    <group android:checkableBehavior="single">
        <item
            android:id="@+id/mi_Cred"
            android:title="红色"/>
        <item
            android:id="@+id/mi_Cblue"
            android:title="蓝色"/>
        <item
            android:id="@+id/mi_Cgreen"
            android:title="绿色"/>
    </group>

</menu>
  1. 在 Activity 中加载 上下文菜单
public void onCreateContextMenu(ContextMenu menu, View view, ContextMenu.ContextMenuInfo info)
{
    getMenuInflater().inflate(R.menu.context_menu,menu);
    super.onCreateContextMenu(menu,view,info);
}
  1. 绑定 View
registerForContextMenu(mListView);
  1. 注册点击事件
// 重载 onContextItemSelected 函数来注册事件响应
public boolean onContextItemSelected(MenuItem item)
{
    item.setChecked(true);
    AdapterView.AdapterContextMenuInfo info=(AdapterView.AdapterContextMenuInfo)(item.getMenuInfo());

    int selected=0;
    for(int i=0;i<mListView.getCount();i++)
    {
        if(mListView.getItemIdAtPosition(i)==info.id)
        {
            selected=i;
            break;
        }
    }

    String str=mListView.getItemAtPosition(selected).toString();

    ((TextView)findViewById(R.id.tv_Title)).setText(str+item.getTitle().toString());
    return true;
}
