# 使用第三方控件

  1. 直接引用 在build.gradle(Module:app)文件的dependency闭包部分，添加相关的引用。以下示例引用的是citypickerview控件。添加后，要注意update gradle
compile 'liji.libiary.dev:citypickerview:0.7.0'
  1. 下载 jar 包来导入
  2. 通过添加module的方式来导入 参考网址：https://www.jianshu.com/p/dba47464fcfc

dependencies{

# 自定义单一控件

  1. 在对应的包下创建继承 于 View 的自定义类
package com.example.testcustomview;

import android.content.Context;
import android.graphics.Canvas;
import android.graphics.Color;
import android.graphics.Paint;
import android.util.AttributeSet;
import android.view.View;

import androidx.appcompat.widget.AppCompatTextView;

public class ChessBoardView extends View
{
    Paint m_paint;
    // 默认的构造函数
    public ChessBoardView(Context context)
    {
        super(context);
        init();
    }

    // 如果需要传入属性的构造函数
    public ChessBoardView(Context context, AttributeSet attrs)
    {
        super(context,attrs);
        init();
    }

    // 由于涉及控件的风格、形状等的修改需要重载 onDraw() 函数
    protected void onDraw(Canvas canvas)
    {
        super.onDraw(canvas);
        int width=getWidth();
        int height=getHeight();

        // 填充
        m_paint.setStyle(Paint.Style.FILL);

        // 设置填充颜色为白色
        m_paint.setColor(Color.WHITE);

        // 画方格子
        canvas.drawRect(0,0,width,height,m_paint);

        //  描边
        m_paint.setStyle(Paint.Style.STROKE);

        // 设置描边的颜色为黑色
        m_paint.setColor(Color.BLACK);

        int cols=6;

        // 画横线
        int size= width/cols; // 每一个小块的大小
        int rows=height / (width / cols)+3;
        for(int r = 0 ; r < rows ; r++)
        {
            int y = r * size;
            canvas.drawLine(0 , y , width , y , m_paint);
        }

        // 画竖线
        for(int c = 0 ; c < cols ; c++)
        {
            int x = c* size;
            canvas.drawLine(x , 0 , x , height , m_paint);
        }


    }

    private void init()
    {
        m_paint=new Paint();
        m_paint.setStrokeWidth(3);
    }

}
  1. 自定义控件的属性 在 res/values 文件夹下新建一个 attr.xml 文件 在resourse下新建一组属性：
<?xml version="1.0" encoding="utf-8"?>
<resources>
    <declare-styleable name="ChessBoardView">
        <attr name="strokeSize" format="integer"/>
        <attr name="bkColor" format="color"/>
    </declare-styleable>
</resources>

为了使属性生效，还要在自定义类中通过 TypedArray类 获取属性值，并且调用 recycle 函数结束。

Paint m_paint;
int m_strokeSize;
int m_bkColor;

// 默认的构造函数
public ChessBoardView(Context context)
{
    super(context);
    init();
}

// 如果需要传入属性的构造函数
public ChessBoardView(Context context, AttributeSet attrs)
{
    super(context,attrs);

    // 设置自定义属性

    // 定义 TypedArray 对象获取 view 的属性组
    TypedArray ta=context.obtainStyledAttributes(attrs,R.styleable.ChessBoardView);

    // 使用 getInteger 等函数获得属性值 以改组属性名 加 下划线和 具体属性开头 第二个参数使缺省值
    m_strokeSize=ta.getInteger(R.styleable.ChessBoardView_strokeSize,3);
    m_bkColor=ta.getColor(R.styleable.ChessBoardView_bkColor,Color.WHITE);

    ta.recycle();

    init();
}

// 由于涉及控件的风格、形状等的修改需要重载 onDraw() 函数
protected void onDraw(Canvas canvas)
{
    super.onDraw(canvas);
    int width=getWidth();
    int height=getHeight();

    // 填充
    m_paint.setStyle(Paint.Style.FILL);

    // 设置填充颜色为 m_bkColor 设定的颜色
    m_paint.setColor(m_bkColor);


    // 画背景
    canvas.drawRect(0,0,width,height,m_paint);



    //  描边
    m_paint.setStyle(Paint.Style.STROKE);

    // 设置描边的颜色为黑色
    m_paint.setColor(Color.BLACK);

    // 画线的颜色和背景色不能一样
    if(m_bkColor==Color.BLACK)
    {
        m_paint.setColor(Color.WHITE);
    }

    // 设置描边的线条粗细
    m_paint.setStrokeWidth(m_strokeSize);

    int cols=6;

    // 画横线
    int size= width/cols; // 每一个小块的大小
    int rows=height / (width / cols)+3;
    for(int r = 0 ; r < rows ; r++)
    {
        int y = r * size;
        canvas.drawLine(0 , y , width , y , m_paint);
    }

    // 画竖线
    for(int c = 0 ; c < cols ; c++)
    {
        int x = c* size;
        canvas.drawLine(x , 0 , x , height , m_paint);
    }


}

  1. 完整代码：
package com.example.testcustomview;

import android.content.Context;
import android.content.res.TypedArray;
import android.graphics.Canvas;
import android.graphics.Color;
import android.graphics.Paint;
import android.util.AttributeSet;
import android.view.View;

import androidx.appcompat.widget.AppCompatTextView;

public class ChessBoardView extends View
{
    Paint m_paint;
    int m_strokeSize;
    int m_bkColor;
    String m_text;

    // 默认的构造函数
    public ChessBoardView(Context context)
    {
        super(context);
        init();
    }

    // 如果需要传入属性的构造函数
    public ChessBoardView(Context context, AttributeSet attrs)
    {
        super(context,attrs);

        // 设置自定义属性

        // 定义 TypedArray 对象获取 view 的属性组
        TypedArray ta=context.obtainStyledAttributes(attrs,R.styleable.ChessBoardView);

        // 使用 getInteger 等函数获得属性值 以改组属性名 加 下划线和 具体属性开头 第二个参数使缺省值
        m_strokeSize=ta.getInteger(R.styleable.ChessBoardView_strokeSize,3);
        m_bkColor=ta.getColor(R.styleable.ChessBoardView_bkColor,Color.WHITE);
        m_text=ta.getString(R.styleable.ChessBoardView_topText);

        ta.recycle();

        init();
    }

    // 由于涉及控件的风格、形状等的修改需要重载 onDraw() 函数
    protected void onDraw(Canvas canvas)
    {
        super.onDraw(canvas);
        int width=getWidth();
        int height=getHeight();

        // 填充
        m_paint.setStyle(Paint.Style.FILL);

        // 设置填充颜色为 m_bkColor 设定的颜色
        m_paint.setColor(m_bkColor);


        // 画背景
        canvas.drawRect(0,width/6,width,height,m_paint);



        //  描边
        m_paint.setStyle(Paint.Style.STROKE);

        // 设置描边的颜色为黑色
        m_paint.setColor(Color.BLACK);

        // 画线的颜色和背景色不能一样
        if(m_bkColor==Color.BLACK)
        {
            m_paint.setColor(Color.WHITE);
        }

        // 设置描边的线条粗细
        m_paint.setStrokeWidth(m_strokeSize);

        int cols=6;

        // 画横线
        int size= width/cols; // 每一个小块的大小
        int rows=height / (width / cols)+3;
        for(int r = 0 ; r < rows ; r++)
        {
            int y = r * size;
            canvas.drawLine(0 , y , width , y , m_paint);
        }

        // 画竖线
        for(int c = 0 ; c < cols ; c++)
        {
            int x = c* size;
            canvas.drawLine(x , width/cols , x , height , m_paint);
        }

        m_paint.setTextSize(100); // 设置文字的大小
        m_paint.setColor(Color.BLACK); // 设置文字的颜色为黑色

        m_paint.setStyle(Paint.Style.FILL); // 设置绘制方式为填充文字
        m_paint.setTextAlign(Paint.Align.CENTER); // 设置水平居中
        canvas.drawText(m_text,width/2,175,m_paint); // 绘制文字



    }

    private void init()
    {
        m_paint=new Paint();
        m_paint.setStrokeWidth(3);
    }



}
